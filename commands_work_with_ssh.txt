# 1. Cập nhật hệ thống và cài đặt Python
# Cập nhật danh sách gói
sudo apt-get update -y

# Cài đặt Python3, Pip và các công cụ hỗ trợ
sudo apt-get install -y python3 python3-pip python3-venv wget

# 2. Thiết lập Virtual Environment
# Tạo môi trường ảo tên là 'mage-env'
python3 -m venv mage-env

# Kích hoạt môi trường ảo
source mage-env/bin/activate

# 3. Cài đặt Mage và các thư viện 
# Cập nhật pip lên bản mới nhất trong venv
pip install --upgrade pip

# Tải các gói trong file requirements
nano requirements.txt

# then paste requirements below, -> ^O -> ^X
pip install -r requirements.txt --force-reinstall

# Final
mage start [my_project_name] (ex: mage start tlc_project)
